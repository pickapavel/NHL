// 🔥 ELO výpočet function calculateElo(oldRating, opponentRating, result){ const K = 40 const expected = 1 / (1 + Math.pow(10, (opponentRating - oldRating)/400)) return oldRating + K * (result - expected) } // 🔥 hlavní funkce pro zápas async function updateMatchElo(matchId){ const { data: match } = await supabase .from("matches") .select("*") .eq("id", matchId) .single() if(!match.home_score || !match.away_score){ console.log("Zápas nemá výsledek") return } const { data: teams } = await supabase .from("teams") .select("*") .in("id", [match.home_team, match.away_team]) const home = teams.find(t=>t.id === match.home_team) const away = teams.find(t=>t.id === match.away_team) const hg = Number(match.home_score) const ag = Number(match.away_score) let stats = null try{ stats = match.stats ? JSON.parse(match.stats) : null }catch(e){} const isSN = stats && (stats[20] === "1" || stats[20] === "2") const isOT = stats && !isSN && (stats[9] === "11" || stats[19] === "11") let homeResult, awayResult // 🔥 základní výsledek if(hg > ag){ homeResult = 1 awayResult = 0 } else if(ag > hg){ homeResult = 0 awayResult = 1 } else{ homeResult = 0.5 awayResult = 0.5 } // 🔥 úprava pro OT/SN if(isOT || isSN){ if(hg > ag){ homeResult = 0.75 awayResult = 0.25 }else{ homeResult = 0.25 awayResult = 0.75 } } // 🔥 nové ratingy const newHome = calculateElo(home.rating_form, away.rating_form, homeResult) const newAway = calculateElo(away.rating_form, home.rating_form, awayResult) // 🔥 uložení do DB await supabase.from("teams") .update({ rating_form: newHome }) .eq("id", home.id) await supabase.from("teams") .update({ rating_form: newAway }) .eq("id", away.id) console.log("ELO updated", newHome, newAway) }
